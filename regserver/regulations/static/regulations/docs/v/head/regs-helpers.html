            /* if we have a supplement */
            interpIndex = $.inArray('Interp', parts);
            if (interpIndex >= 0) {
                return this.interpId(parts.slice(0, interpIndex));
            }
            /* if we have an appendix */
            if (isNaN(parseInt(parts[1], 10))) {
                return this.appendixId(parts[0], parts[1]);

            for (i = 0; i <= len; i++) {

                    ref += parts[i];
                }

                }
            }
